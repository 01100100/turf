Module: turf/along

Returns a point at a specified distance along a line.
Parameters:
Name Type Attributes Default Description
Line LineString to move along
Distance Number to move
units String <optional>
 miles can be degrees, radians, miles, or kilometers
Source:
  • node_modules/turf-along/index.js, line 6
Returns:
Point along the line at X distance
Type
Point

var line = {
  "type": "Feature",
  "properties": {},
  "geometry": {
    "type": "LineString",
    "coordinates": [
      [
        -77.0316696166992,
        38.878605901789236
      ],
      [
        -77.02960968017578,
        38.88194668656296
      ],
      [
        -77.02033996582031,
        38.88408470638821
      ],
      [
        -77.02566146850586,
        38.885821800123196
      ],
      [
        -77.02188491821289,
        38.88956308852534
      ],
      [
        -77.01982498168944,
        38.89236892551996
      ]
    ]
  }
}

var along = turf.along(line, 1, 'miles');
